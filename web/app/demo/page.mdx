```jsx
// title app/page.js
import { Code } from "bright"
import { SomeContext, Input, Demo } from "./client-components"

Code.extensions = {
  input: {
    InlineAnnotation: ({ content, query }) => (
      <Input defaultValue={content} name={query} />
    ),
  },
}

const code = `
<div
  style={{
    // input[13:19] color
    color: "#fafafa",
    // input[18:24] bg
    background: "#222222",
  }}
>
  Hello World
</div>`

export default function Page() {
  return (
    <SomeContext>
      <Code lang="js">{code}</Code>
      <Demo />
    </SomeContext>
  )
}
```
